    private VBox createFormLayout() {

            -fx-font-family: Arial;

        areaArea.setMaxHeight(50);

        // Button to submit
        Button submitButton = new Button("Gửi");
        submitButton.setFont(Font.font("Arial", 16));
        submitButton.setOnAction(e -> {
            // Lấy dữ liệu từ các trường
            String name = nameField.getText();
            String message = messageArea.getText();
            String area = areaArea.getText();

            // Kiểm tra nếu có trường nào còn trống
            if (name.isEmpty() || message.isEmpty() || area.isEmpty()) {
                // Hiển thị thông báo lỗi nếu có trường trống
                showAlert(Alert.AlertType.ERROR, "Lỗi", "Vui lòng điền đầy đủ thông tin.");
            } else {
                // Nếu mọi thứ đều hợp lệ, hiển thị thông báo thành công
                System.out.println("Tên: " + name);
                System.out.println("Lời nhắn: " + message);
                System.out.println("Địa điểm bạn muốn chúng tôi đến: " + area);

                // Hiển thị thông báo thành công
                showAlert(Alert.AlertType.INFORMATION, "Thông báo", "Bạn đã gửi thông điệp thành công.");
            }

            // Làm mới các trường nhập liệu sau khi gửi
            nameField.clear();
            messageArea.clear();
            areaArea.clear();
        });


        formLayout.getChildren().addAll(nameLabel, nameField, messageLabel, messageArea, areaLabel, areaArea, submitButton);
        return formLayout;
    }
    private void showAlert(Alert.AlertType alertType, String title, String message) {
        Alert alert = new Alert(alertType);
        alert.setTitle(title);
        alert.setHeaderText(null);
        alert.setContentText(message);
        alert.showAndWait();
    }

    private void cycleImages(HBox studentGallery) {

        String[][] imageSets = {
                {"2.jpg", "3.jpg", "6.jpg", "Nguyễn Song Nhi", "Châu Quỳnh Như", "Huỳnh Ngọc Nga"},
                {"7.jpg", "9.jpg", "12.jpg", "Nguyễn Lê Quỳnh", "Nguyễn Hà My", "Lê Huỳnh Như"}
        };


        String[] currentSet = imageSets[imageSetIndex];

        for (int i = 0; i < 3; i++) {
            VBox studentCard = (VBox) studentGallery.getChildren().get(i);
            ImageView imageView = (ImageView) studentCard.getChildren().get(0);
            imageView.setImage(new Image(currentSet[i]));
            Label nameLabel = (Label) studentCard.getChildren().get(1);
            nameLabel.setText(currentSet[i + 3]);
        }

        // Update the image set index
        imageSetIndex = (imageSetIndex + 1) % imageSets.length;
    }

    private StackPane createImagePane(String imagePath, String title, String description, String imageBackgroundColor, String contentBackgroundColor) {
        // Tạo StackPane cho mỗi hình ảnh
        StackPane stackPane = new StackPane();
        stackPane.setPrefWidth(300);
        stackPane.setPrefHeight(300);


        // Tạo ImageView cho ảnh
        ImageView imageView = new ImageView(new Image(imagePath));
        imageView.setFitWidth(300);  // Chiều rộng cố định
        imageView.setFitHeight(300);  // Chiều cao cố định
        imageView.setPreserveRatio(true);  // Giữ tỷ lệ gốc của ảnh
        imageView.setSmooth(true);  // Làm ảnh mượt mà
        imageView.setCache(true);  // Tăng hiệu suất khi xử lý ảnh


        // Tạo mặt nội dung (mặt trước khi lật)
        Rectangle frontRectangle = new Rectangle(400, 400);
        frontRectangle.setFill(Color.web(contentBackgroundColor));  // Dùng mã màu cho nền


        // Tạo tiêu đề (chữ lớn)
        Label titleLabel = new Label(title);
        titleLabel.setStyle("-fx-font-size: 28px; -fx-font-weight: bold; -fx-text-fill: #FFFFFF;");
        titleLabel.setAlignment(Pos.CENTER);


        // Tạo nội dung mô tả (chữ nhỏ hơn)
        Label descriptionLabel = new Label(description);
        descriptionLabel.setWrapText(true);  // Cho phép xuống dòng
        descriptionLabel.setMaxWidth(350);  // Đặt chiều rộng tối đa để gọn trong khung
        descriptionLabel.setStyle("-fx-font-size: 18px; -fx-text-fill: #FFFFFF; -fx-text-alignment: center;"); // Căn trái
        descriptionLabel.setAlignment(Pos.TOP_LEFT);  // Căn trên và trái


        // Gộp tiêu đề và nội dung vào VBox
        VBox contentBox = new VBox(10);  // Khoảng cách giữa các phần tử
        contentBox.setAlignment(Pos.CENTER);  // Căn giữa theo chiều dọc
        contentBox.getChildren().addAll(titleLabel, descriptionLabel);


        // Gộp mặt nội dung vào một StackPane
        StackPane contentStack = new StackPane();
        contentStack.getChildren().addAll(frontRectangle, contentBox);


        // Đặt màu nền của khung StackPane
        stackPane.setStyle("-fx-background-color: " + imageBackgroundColor + ";");


        // Thêm ảnh vào StackPane
        stackPane.getChildren().add(imageView);


        // Hiển thị nội dung khi di chuột vào ảnh
        stackPane.setOnMouseEntered(event -> {
            stackPane.getChildren().clear();  // Xóa ảnh
            stackPane.getChildren().add(contentStack);  // Hiển thị mặt nội dung
        });


        // Hiển thị lại ảnh khi chuột rời
        stackPane.setOnMouseExited(event -> {
            stackPane.getChildren().clear();  // Xóa mặt nội dung
            stackPane.getChildren().add(imageView);  // Hiển thị lại ảnh
        });

        return stackPane;
    }
