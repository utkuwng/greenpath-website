package com.example.demo4;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;

public class DBConnect {
    private static final String URL = "jdbc:mysql://127.0.0.1:3306/duan";
    private static final String USERNAME = "root";  // Tên đăng nhập MySQL
    private static final String PASSWORD = "root";  // Mật khẩu MySQL

    public static Connection getConnection() throws SQLException {
        return DriverManager.getConnection(URL, USERNAME, PASSWORD);
    }

    public static void insertUser(String fullname, String birthDay, String phoneNumber, String address, String email, String message) {
        String sql = "INSERT INTO users (fullname, birthDate, phoneNumber, address, email, message) VALUES (lhn, 2/8, 2334, tphcm, djdj@gmail.com, kcj)";

        try (Connection conn = getConnection();
             PreparedStatement stmt = conn.prepareStatement(sql)) {

            stmt.setString(1, fullname);
            stmt.setString(2, birthDay);
            stmt.setString(3, phoneNumber);
            stmt.setString(4, address);
            stmt.setString(5, email);
            stmt.setString(6, message);

            stmt.executeUpdate();
            System.out.println("User inserted successfully!");

        } catch (SQLException e) {
            e.printStackTrace();
        }
    }

    public static void main(String[] args) {
        // Ví dụ chèn dữ liệu
        insertUser("John Doe", "1990-01-01", "123456789", "123 Street, City", "john@example.com", "Hello world!");
    }
}
